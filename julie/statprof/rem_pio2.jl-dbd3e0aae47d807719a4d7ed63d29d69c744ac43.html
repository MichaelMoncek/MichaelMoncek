98 """
99 fromfraction(f::Int128)
100 Compute a tuple of values `(z1,z2)` such that
101 ``z1 + z2 == f / 2^128``
102 and the significand of `z1` has 27 trailing zeros.
103 """
104 function fromfraction(f::Int128)
105 if f == 0
106 return (0.0,0.0)
107 end
108
109 # 1. get leading term truncated to 26 bits
110 s = ((f < 0) % UInt64) << 63 # sign bit
111 x = abs(f) % UInt128 # magnitude
112 n1 = Base.top_set_bit(x) # ndigits0z(x,2)
113 m1 = ((x >> (n1-26)) % UInt64) << 27
114 d1 = ((n1-128+1021) % UInt64) << 52
115 z1 = reinterpret(Float64, s | (d1 + m1))
116
117 # 2. compute remaining term
118 x2 = (x - (UInt128(m1) << (n1-53)))
119 if x2 == 0
120 return (z1, 0.0)
121 end
122 n2 = Base.top_set_bit(x2)
123 m2 = (x2 >> (n2-53)) % UInt64
124 d2 = ((n2-128+1021) % UInt64) << 52
125 z2 = reinterpret(Float64, s | (d2 + m2))
126 return (z1,z2)
127 end

129 function paynehanek(x::Float64)
130 # 1. Convert to form
131 #
132 # x = X * 2^k,
133 #
134 # where 2^(n-1) <= X < 2^n is an n-bit integer (n = 53, k = exponent(x)-52 )
135
136 # Computations are integer based, so reinterpret x as UInt64
137 u = reinterpret(UInt64, x)
138 # Strip x of exponent bits and replace with ^1
139 X = (u & significand_mask(Float64)) | (one(UInt64) << significand_bits(Float64))
140 # Get k from formula above
141 # k = exponent(x)-52
142 raw_exponent = ((u & exponent_mask(Float64)) >> significand_bits(Float64)) % Int
143 k = raw_exponent - exponent_bias(Float64) - significand_bits(Float64)
144
145 # 2. Let α = 1/2π, then:
146 #
147 # α*x mod 1 ≡ [(α*2^k mod 1)*X] mod 1
148 #
149 # so we can ignore the first k bits of α. Extract the next 3 64-bit parts of α.
150 #
151 # i.e. equivalent to
152 # setprecision(BigFloat,4096)
153 # α = 1/(2*big(pi))
154 # A = mod(ldexp(α,k), 1)
155 # z1 = ldexp(A,64)
156 # a1 = trunc(UInt64, z1)
157 # z2 = ldexp(z1-a1, 64)
158 # a2 = trunc(UInt64, z2)
159 # z3 = ldexp(z2-a2, 64)
160 # a3 = trunc(UInt64, z3)
161
162 # This is equivalent to
163 # idx, shift = divrem(k, 64)
164 # but divrem is slower.
165 idx = k >> 6
166
167 shift = k - (idx << 6)
168 Base.@assume_effects :nothrow :noub @inbounds if shift == 0
169 a1 = INV_2PI[idx+1]
170 a2 = INV_2PI[idx+2]
171 a3 = INV_2PI[idx+3]
172 else
173 # use shifts to extract the relevant 64 bit window
174 a1 = (idx < 0 ? zero(UInt64) : INV_2PI[idx+1] << shift) | (INV_2PI[idx+2] >> (64 - shift))
175 a2 = (INV_2PI[idx+2] << shift) | (INV_2PI[idx+3] >> (64 - shift))
176 a3 = (INV_2PI[idx+3] << shift) | (INV_2PI[idx+4] >> (64 - shift))
177 end
178
179 # 3. Perform the multiplication:
180 #
181 # X. 0 0 0
182 # × 0. a1 a2 a3
183 # ==============
184 # _. w w _
185 #
186 # (i.e. ignoring integer and lowest bit parts of result)
187
188 w1 = UInt128(X*a1) << 64 # overflow becomes integer
189 w2 = widemul(X,a2)
190 w3 = widemul(X,a3) >> 64
191 w = w1 + w2 + w3 # quotient fraction after division by 2π
192
193 # adjust for sign of x
194 w = flipsign(w,x)
195
196 # 4. convert to quadrant, quotient fraction after division by π/2:
197 q = (((w>>125)%Int +1)>>1) # nearest quadrant
198 f = (w<<2) % Int128 # fraction part of quotient after division by π/2, taking values on [-0.5,0.5)
199
200 # 5. convert quotient fraction to split precision Float64
201 z_hi,z_lo = fromfraction(f)
202
203 # 6. multiply by π/2
204 pio2 = 1.5707963267948966
205 pio2_hi = 1.5707963407039642
206 pio2_lo = -1.3909067614167116e-8
207 y_hi = (z_hi+z_lo)*pio2
208 y_lo = (((z_hi*pio2_hi - y_hi) + z_hi*pio2_lo) + z_lo*pio2_hi) + z_lo*pio2_lo
209 return q, DoubleFloat64(y_hi, y_lo)
210 end
