7 Read a PNG image as a julia `Array`.
8
9 # Arguments
10
11 - `fpath`: the path to the png file to be loaded.
12 - `s`: an IO stream containing the png file.
13
14 # Keywords
15
16 - `gamma`: the end-to-end coefficient for gamma correction, can be used to override the
17 automatic gamma correction, a value of `1.0` means no gamma correction. By default,
18 gamma correction is applied according to a `gAMA` or `sRGB` chunks, if present;
19 if neither chunk is present a value of `0.45455` for the file gamma is assumed.
20 Screen gamma is currently always assumed to be `2.2`.
21 - `background`: can be used to set a background color for transparent images.
22 Accepted values are `false` for no background, `true` for loading the image
23 with a solid background if `bKGD` chunk is set, as well as user-provided backgrounds
24 of type `UInt8`, `Gray` or `RGB`, which represent a palette index, gray and true color
25 backgrounds respectively.
26 - `expand_paletted`: when reading in simple paletted images, i.e. having a `PLTE` chunk and
27 an 8 bit depth, the image will be represented as an `IndirectArray` with `OffsetArray`
28 `values` field. To always get back a plain `Matrix` of colorants, use `expand_paletted=true`.
29
30 # Returns
31
32 - `Matrix{Gray{N0f8}}`: for grayscale images (without transparency) and bit depth lower or equal to 8.
33 - `Matrix{Gray{N0f16}}`: for grayscale images (without transparency) and bit depth equal to 16.
34 - `Matrix{GrayA{N0f8}}`: for grayscale images (with transparency) and bit depth lower or equal to 8.
35 - `Matrix{GrayA{N0f16}}`: for grayscale images (with transparency) and bit depth equal to 16.
36 - `Matrix{RGB{N0f8}}`: for true color images (without transparency) and bit depth equal to 8.
37 Also for palleted images without transparency when `expand_paletted=true`.
38 - `Matrix{RGB{N0f16}}`: for true color images (without transparency) and bit depth equal to 16.
39 - `Matrix{RGBA{N0f8}}`: for true color images (with transparency) and bit depth equal to 8.
40 Also for palleted images without transparency when `expand_paletted=true`.
41 - `Matrix{RGBA{N0f16}}`: for true color images (with transparency) and bit depth equal to 16.
42 - `IndirectArray{RGB{N0f8}, 2, UInt8, Matrix{UInt8}, OffsetVector{RGB{N0f8}, Vector{RGB{N0f8}}}}`:
43 for palleted images with transparency when `expand_paletted=false`.
44 - `IndirectArray{RGBA{N0f8}, 2, UInt8, Matrix{UInt8}, OffsetVector{RGBA{N0f8}, Vector{RGBA{N0f8}}}}`:
45 for palleted images with transparency when `expand_paletted=false`.

105 function _load(png_ptr, info_ptr; gamma::Union{Nothing,Float64}=nothing, expand_paletted::Bool=false, background=false)
106 png_read_info(png_ptr, info_ptr)
107 flags = png_get_valid(
108 png_ptr, info_ptr,
109 PNG_INFO_sRGB | PNG_INFO_gAMA | PNG_INFO_tRNS | PNG_INFO_cHRM | PNG_INFO_bKGD | PNG_INFO_PLTE
110 )
111 valid_sRGB = (flags & PNG_INFO_sRGB) != 0
112 valid_gAMA = (flags & PNG_INFO_gAMA) != 0
113 valid_tRNS = (flags & PNG_INFO_tRNS) != 0
114 valid_cHRM = (flags & PNG_INFO_cHRM) != 0
115 valid_bKGD = (flags & PNG_INFO_bKGD) != 0
116 valid_PLTE = (flags & PNG_INFO_PLTE) != 0
117
118 width = png_get_image_width(png_ptr, info_ptr)
119 height = png_get_image_height(png_ptr, info_ptr)
120 color_type = color_type_orig = png_get_color_type(png_ptr, info_ptr)
121 bit_depth = bit_depth_orig = png_get_bit_depth(png_ptr, info_ptr)
122 num_channels = png_get_channels(png_ptr, info_ptr)
123 interlace_type = png_get_interlace_type(png_ptr, info_ptr)
124 background_color = nothing
125 is_transparent = valid_tRNS | (color_type & PNG_COLOR_MASK_ALPHA) != 0
126 is_gray = (color_type == PNG_COLOR_TYPE_GRAY_ALPHA) | (color_type == PNG_COLOR_TYPE_GRAY)
127
128 read_as_paletted = !expand_paletted && color_type == PNG_COLOR_TYPE_PALETTE && bit_depth == 8 && valid_PLTE
129 read_with_background = _check_background_load(background, is_gray, valid_PLTE, read_as_paletted)
130
131 screen_gamma = PNG_DEFAULT_sRGB
132 image_gamma = Ref{Cdouble}(-1.0)
133 intent = Ref{Cint}(-1)
134 if isnothing(gamma)
135 if valid_sRGB
136 if png_get_sRGB(png_ptr, info_ptr, intent) != 0
137 png_set_gamma(png_ptr, screen_gamma, PNG_DEFAULT_sRGB)
138 else
139 if png_get_gAMA(png_ptr, info_ptr, image_gamma) == 0
140 image_gamma[] = 0.45455
141 end
142 png_set_gamma(png_ptr, screen_gamma, image_gamma[])
143 end
144 elseif valid_gAMA
145 if png_get_gAMA(png_ptr, info_ptr, image_gamma) == 0
146 image_gamma[] = 0.45455
147 end
148 png_set_gamma(png_ptr, screen_gamma, image_gamma[])
149 end
150 elseif gamma != 1
151 image_gamma[] = 2.2inv(gamma)
152 png_set_gamma(png_ptr, screen_gamma, image_gamma[])
153 end
154
155 if !read_as_paletted
156 if color_type == PNG_COLOR_TYPE_PALETTE
157 png_set_palette_to_rgb(png_ptr)
158 color_type = PNG_COLOR_TYPE_RGB
159 end
160
161 if color_type == PNG_COLOR_TYPE_GRAY && bit_depth < 8
162 png_set_expand_gray_1_2_4_to_8(png_ptr)
163 png_set_packing(png_ptr)
164 bit_depth = 8
165 end
166
167 if valid_tRNS

169 if color_type == PNG_COLOR_TYPE_GRAY || color_type == PNG_COLOR_TYPE_RGB
170 color_type |= PNG_COLOR_MASK_ALPHA
171 end
172 end
173
174 if read_with_background & valid_bKGD & is_transparent
175 png_set_filler(png_ptr, 0xff, PNG_FILLER_AFTER)
176 end
177 buffer_eltype = _buffer_color_type(color_type, bit_depth)
178
179 bit_depth == 16 && png_set_swap(png_ptr)
180 else
181 buffer_eltype = UInt8
182 end
183
184 if read_with_background
185 background_color = process_background(png_ptr, info_ptr, _adjust_background_bitdepth(background, bit_depth))
186 is_transparent || @warn("Background color for non-transparent image: $(background_color)")
187 end
188
189 n_passes = png_set_interlace_handling(png_ptr)
190 png_read_update_info(png_ptr, info_ptr)
191
192 # Gamma correction is applied to a palette after `png_read_update_info` is called
193 if read_as_paletted
194 palette_length = Ref{Cint}(0)
195 palette_buffer = Ref{Ptr{PNGFiles.png_color_struct}}(C_NULL)
196 png_get_PLTE(png_ptr, info_ptr, palette_buffer, palette_length)
197 palette = Vector{RGB{N0f8}}(undef, palette_length[])
198 GC.@preserve palette unsafe_copyto!(pointer(palette), Ptr{RGB{N0f8}}(palette_buffer[]), length(palette))
199 if valid_tRNS
200 alpha_buffer = Ref{Ptr{UInt8}}(C_NULL)
201 alphas_cnt = Ref{Cint}(0)
202 png_get_tRNS(png_ptr, info_ptr, alpha_buffer, alphas_cnt, C_NULL)
203 alpha = fill(one(N0f8), palette_length[]) # palette entries are opaque by default
204 @assert palette_length[] >= alphas_cnt[]
205 GC.@preserve alpha unsafe_copyto!(pointer(alpha), Ptr{N0f8}(alpha_buffer[]), min(palette_length[], alphas_cnt[]))
206 palette = map(RGBA, palette, alpha)
207 end
208 buffer_eltype = UInt8
209 end
210
211 # We transpose to work around libpng expecting row-major arrays
212 buffer = Matrix{buffer_eltype}(undef, width, height)
213
214 @debug(
215 "Read PNG info:",
216 PNG_HEADER_VERSION_STRING,
217 png_ptr,
218 height,
219 width,
220 color_type_orig,
221 color_type,
222 bit_depth_orig,
223 bit_depth,
224 num_channels,
225 interlace_type,
226 gamma,
227 image_gamma[],
228 screen_gamma,
229 background,
230 intent[],
231 read_as_paletted,
232 n_passes,
233 buffer_eltype,
234 valid_sRGB,
235 valid_gAMA,
236 valid_tRNS,
237 valid_cHRM,
238 valid_bKGD,
239 valid_PLTE,
240 )
241
242 GC.@preserve buffer begin
243 buffer = Base.invokelatest(_load!, buffer, png_ptr, info_ptr)
244 end
245
246 if expand_paletted || color_type != PNG_COLOR_TYPE_PALETTE
247 return buffer
248 else
249 # We got 0-based indices back from libpng and converting to 1-based could overflow UInt8.
250 # Using UInt16 for index would cost us large part of the savings provided by IndirectArray.
251 return IndirectArray(buffer, OffsetArray(palette, -1))
252 end

262 function _buffer_color_type(color_type, bit_depth)
263 bit_depth = Int(bit_depth)
264 if color_type == PNG_COLOR_TYPE_GRAY
265 colors_type = Gray{bit_depth > 8 ? Normed{UInt16,bit_depth} : Normed{UInt8,bit_depth}}
266 elseif color_type == PNG_COLOR_TYPE_PALETTE
267 colors_type = RGB{bit_depth == 16 ? N0f16 : N0f8}
268 elseif color_type == PNG_COLOR_TYPE_RGB
269 colors_type = RGB{bit_depth == 16 ? N0f16 : N0f8}
270 elseif color_type == PNG_COLOR_TYPE_RGB_ALPHA
271 colors_type = RGBA{bit_depth == 16 ? N0f16 : N0f8}
272 elseif color_type == PNG_COLOR_TYPE_GRAY_ALPHA
273 colors_type = GrayA{bit_depth > 8 ? Normed{UInt16,bit_depth} : Normed{UInt8,bit_depth}}
274 else
275 throw(error("Unknown color type: $color_type"))
276 end
277 return colors_type

305 Write out a julia `Array` as a PNG image.
306
307 # Arguments
308 - `fpath`: the filesystem path where to store the resulting PNG image.
309 - `s`: the IO stream to which the resulting PNG image should be written to.
310 - `image`: the julia `Array` which should be encoded as a PNG image. The type, eltype, and
311 dimensionality jointly determine the way the PNG file is encoded. The following table
312 roughly summarises the encoding rules:
313 | **dims** | **type** | **eltype** (`T`) | **PNG color type** | **PNG bit depth** |
314 |:------------|:-----------------------------------------------|:-----------------------------------------------------|:----------------------------|-------------------------------------------------:|
315 | `[h, w]` | `AbstracArray{T,2}` | `Union{AbstractFloat, Unsigned, Normed, Gray}` | `PNG_COLOR_TYPE_GRAY` | 8 by default, 16 when `N0f16` or `N4f12` is used |
316 | `[h, w, 1]` | `AbstracArray{T,3}` | `Union{AbstractFloat, Unsigned, Normed, Gray}` | `PNG_COLOR_TYPE_GRAY` | 8 by default, 16 when `N0f16` or `N4f12` is used |
317 | `[h, w, 2]` | `AbstracArray{T,3}` | `Union{AbstractFloat, Unsigned, Normed, GrayA}` | `PNG_COLOR_TYPE_GRAY_ALPHA` | 8 by default, 16 when `N0f16` or `N4f12` is used |
318 | `[h, w, 3]` | `AbstracArray{T,3}` | `Union{AbstractFloat, Unsigned, Normed, RGB, BGR}` | `PNG_COLOR_TYPE_RGB` | 8 by default, 16 when `N0f16` or `N4f12` is used |
319 | `[h, w, 4]` | `AbstracArray{T,3}` | `Union{AbstractFloat, Unsigned, Normed, RGBA, ARGB}` | `PNG_COLOR_TYPE_RGB_ALPHA` | 8 by default, 16 when `N0f16` or `N4f12` is used |
320 | `[h, w]` | `IndirectArray{T, 2, S, Matrix{S}, Vector{T}}` | `Union{RGB, BGR, RGBA, ARGB}` | `PNG_COLOR_TYPE_PALETTE` | 8 |
321
322 ``Note``: Only `UInt8` and `UInt16` are supported for `Unsigned`, `Float32` and `Float64` for `AbstractFloat`;
323 `N0f8`, `N0f16` and `N4f12` for `Normed`. These types are also the only valid types for colorants.
324
325 ``Note``: `Bool`s and `BitArray` are also supported, but the bit depth of the image will still be 8.
326
327 ``Note``: When `image` is an `IndirectArray` with up to 256 unique `RGB` colors, the result is encoded as a paletted image.
328 Palletes with 16 bit depths are not supported. The palette (`values` field of the `IndirectArray`) could also be represented with an `OffsetArray`.
329
330 # Keywords
331

333 - `compression_strategy`: 0 (`Z_DEFAULT_STRATEGY`), 1 (`Z_FILTERED`), 2 (`Z_HUFFMAN_ONLY`),

335 - `filters`: specify a type of preprocessing applied to each row which can increase its compressability.

337 - `file_gamma`: the value governing the gamma encoding of the image. When `nothing`,

339 chunk of the image.
340 - `background`: optional background color to be stored in the `bKGD` chunk. Only meaningful for transparent images.
341 Valid values are `nothing` for no background, `UInt8` as a palette index for palleted images,
342 `Gray` for grayscale images and `RGB` for true color images.
343 - `dpi`: stores the pixel density given in dots per inch into the `pHYs` chunk as pixels per meter.
344 The density is given as `dpi` for ease of use as pixels per meter is an uncommon format. If set to `nothing`,

464 end

589 function __prepare_buffer(x::AbstractArray{T,3}) where {T}
590 nchannels = size(x, 3)
591 if nchannels == 1
592 ifelse(ndims(x) == 3, _prepare_buffer(dropdims(x, dims=3)), x)
593 elseif nchannels == 2
594 GrayA.(colorview(GrayA, view(x, :, :, 1), view(x, :, :, 2)))
595 elseif nchannels == 3
596 RGB.(colorview(RGB, view(x, :, :, 1), view(x, :, :, 2), view(x, :, :, 3)))
597 elseif nchannels == 4
598 RGBA.(colorview(RGBA, view(x, :, :, 1), view(x, :, :, 2), view(x, :, :, 3), view(x, :, :, 4)))
599 else
600 error("Unsupported number of channels $(nchannels) in input. Only <= 4 is expected.")
601 end
602 end

624 function _get_color_type(
625 x::AbstractArray{T, N}
626 ) where {
627 T<:Union{Normed,Unsigned,Bool,AbstractFloat},
628 N
629 }
630 if N == 2
631 return PNG_COLOR_TYPE_GRAY
632 elseif N == 3
633 d = size(x, 3)
634 d == 1 && (return PNG_COLOR_TYPE_GRAY)
635 d == 2 && (return PNG_COLOR_TYPE_GRAY_ALPHA)
636 d == 3 && (return PNG_COLOR_TYPE_RGB)
637 d == 4 && (return PNG_COLOR_TYPE_RGBA)
638 end
639 error("Number of dimensions $(N) in image not supported (only 2D or 3D Arrays are expected).")
640 end
