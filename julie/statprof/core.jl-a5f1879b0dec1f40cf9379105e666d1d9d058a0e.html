151 function write_transport_layer(stream, response)
152 response_utf8 = transcode(UInt8, response)
153 n = length(response_utf8)
154 write(stream, "Content-Length: $n\r\n\r\n")
155 write(stream, response_utf8)
156 flush(stream)
157 end

159 function read_transport_layer(stream)
160 try
161 header_dict = Dict{String,String}()
162 line = chomp(readline(stream))
163 # Check whether the socket was closed
164 if line == ""
165 return nothing
166 end
167 while length(line) > 0
168 h_parts = split(line, ":")
169 header_dict[chomp(h_parts[1])] = chomp(h_parts[2])
170 line = chomp(readline(stream))
171 end
172 message_length = parse(Int, header_dict["Content-Length"])
173 message_str = String(read(stream, message_length))
174 return message_str
175 catch err
176 if err isa Base.IOError
177 return nothing
178 end
179
180 rethrow(err)
181 end
182 end

186 function Base.run(x::JSONRPCEndpoint)
187 x.status == :idle || error("Endpoint is not idle.")
188
189 x.write_task = @async try
190 try
191 for msg in x.out_msg_queue
192 if isopen(x.pipe_out)
193 write_transport_layer(x.pipe_out, msg)
194 else
195 # TODO Reconsider at some point whether this should be treated as an error.
196 break
197 end
198 3 (0 %) 3 (0 %) end
199 finally
200 close(x.out_msg_queue)
201 end
202 catch err
203 bt = catch_backtrace()
204 if x.err_handler !== nothing
205 x.err_handler(err, bt)
206 else
207 Base.display_error(stderr, err, bt)
208 end
209 end
210
211 x.read_task = @async try
212 while true
213 # First we delete any cancellation sources that are no longer needed. We do it this way to avoid a lock
214 while isready(x.no_longer_needed_cancellation_sources)
215 no_longer_needed_cs_id = take!(x.no_longer_needed_cancellation_sources)
216 delete!(x.cancellation_sources, no_longer_needed_cs_id)
217 end
218
219 # Now handle new messages
220 message = read_transport_layer(x.pipe_in)
221
222 if message === nothing || x.status == :closed
223 break
224 end
225
226 message_dict = JSON.parse(message)
227
228 if haskey(message_dict, "method")
229 method_name = message_dict["method"]
230 params = get(message_dict, "params", nothing)
231 id = get(message_dict, "id", nothing)
232 cancel_source = id === nothing ? nothing : CancellationTokens.CancellationTokenSource()
233 cancel_token = cancel_source === nothing ? nothing : CancellationTokens.get_token(cancel_source)
234
235 if method_name == "\$/cancelRequest"
236 id_of_cancelled_request = params["id"]
237 cs = get(x.cancellation_sources, id_of_cancelled_request, nothing) # We might have sent the response already
238 if cs !== nothing
239 CancellationTokens.cancel(cs)
240 end
241 else
242 if id !== nothing
243 x.cancellation_sources[id] = cancel_source
244 end
245
246 request = Request(
247 method_name,
248 params,
249 id,
250 cancel_token
251 )
252
253 try
254 put!(x.in_msg_queue, request)
255 catch err
256 if err isa InvalidStateException
257 break
258 else
259 rethrow(err)
260 end
261 end
262 end
263 else
264 # This must be a response
265 id_of_request = message_dict["id"]
266
267 channel_for_response = x.outstanding_requests[id_of_request]
268 put!(channel_for_response, message_dict)
269 end
270 end
271
272 close(x.in_msg_queue)
273
274 for i in values(x.outstanding_requests)
275 close(i)
276 end
277
278 x.status = :closed
279 catch err
280 bt = catch_backtrace()
281 if x.err_handler !== nothing
282 x.err_handler(err, bt)
283 else
284 Base.display_error(stderr, err, bt)
285 end
286 end
287
288 x.status = :running
289 end

303 function send_request(x::JSONRPCEndpoint, method::AbstractString, params)
304 check_dead_endpoint!(x)
305
306 id = string(UUIDs.uuid4())
307 message = Dict("jsonrpc" => "2.0", "method" => method, "params" => params, "id" => id)
308
309 response_channel = Channel{Any}(1)
310 x.outstanding_requests[id] = response_channel
311
312 message_json = sprint(JSON.show_json, x.serialization, message)
313
314 put!(x.out_msg_queue, message_json)
315
316 response = take!(response_channel)
317
318 if haskey(response, "result")
319 return response["result"]
320 elseif haskey(response, "error")
321 error_code = response["error"]["code"]
322 error_msg = response["error"]["message"]
323 error_data = get(response["error"], "data", nothing)
324 throw(JSONRPCError(error_code, error_msg, error_data))
325 else
326 throw(JSONRPCError(0, "ERROR AT THE TRANSPORT LEVEL", nothing))
327 end
328 end
