37 function enforce_canonical_type(f, data)
38 AT = canonical_type(f, data)
39 # This may not be type stable if `AT` is not a concrete type,
40 # but it's not an issue for `load`; it can never be type stable.
41
42 # work around the invokelatest overhead with an eager type check
43 if data isa AT
44 return data
45 else
46 # the backend might provide its own convert method
47 # use invokelatest to avoid world age issues
48 # See issue #34
49 return Base.invokelatest(convert, AT, data)
50 end
51 end
