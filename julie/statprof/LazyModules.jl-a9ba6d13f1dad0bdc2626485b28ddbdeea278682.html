84 macro lazy(ex)
85 if ex.head in (:import, :using)
86 error("require `@lazy $(ex)=UUID` format.")
87 end
88 if ex.head != :(=)
89 error("unrecognized expression: $(ex)")
90 end
91 uuid = UUID(ex.args[2])
92 ex = ex.args[1]
93
94 if !_LAZYMODE[]
95 @info "disable lazy package loading: environment variable `LazyModules_lazyload=false` is detected" maxlog=1
96 return ex
97 end
98 if ex.head != :import
99 @warn "only `import` command is supported, fallback to eager mode"
100 return ex
101 end
102 args = ex.args
103 if length(args) != 1
104 @warn "only single package import is supported, fallback to eager mode"
105 return ex
106 end
107 x = args[1]
108 if x.head == :.
109 # usage: @lazy import Foo
110 m = _lazy_load(__module__, x.args[1], uuid, x.args[1])
111 # TODO(johnnychen94): the background eager loading seems to work only for Main scope
112 isa(m, Module) && return m
113 isnothing(m) && return ex
114 return m
115 elseif x.head == :(:)
116 # usage: @lazy import Foo: foo, bar
117 @warn "lazily importing symbols are not supported, fallback to eager mode"
118 return ex
119 elseif x.head == :as # compat: Julia at least v1.6
120 # usage: @lazy import Foo as LazyFoo
121 m = _lazy_load(__module__, x.args[2], uuid, x.args[1].args[1])
122 isa(m, Module) && return m
123 isnothing(m) && return ex
124 return m
125 else
126 @warn "unrecognized syntax $ex"
127 return ex
128 end
129 end

131 function _lazy_load(mod, name::Symbol, uuid::UUID, sym::Symbol)
132 if isdefined(mod, name)
133 # otherwise, Revise will constantly trigger the constant redefinition warning
134 m = getfield(mod, name)
135 if m isa LazyModule || m isa Module
136 return m
137 else
138 @warn "Failed to import module, the name `$name` already exists, do nothing"
139 return nothing
140 end
141 end
142 try
143 m = LazyModule(uuid, String(sym))
144 Core.eval(mod, :(const $(name) = $m))
145 return m
